	id = rus.31
	title = rus.31.title
	desc = rus.31.desc
	picture = GFX_doesnt_matter
	hidden = yes
	is_triggered_only = yes
	fire_only_once = yes
	trigger = {
		tag = SOV
		
		NOT = { OR = { has_idea = SOV_governorate_crisis_resolved has_idea = SOV_governorate_crisis_war has_idea = SOV_riots_in_ukraine has_idea = SOV_riots_in_baltics has_idea = SOV_riots_in_belarus has_idea = SOV_riots_in_transcaucasia } }
	}
	immediate = {
		random_list = {
			15 = {
				country_event = { id = rus.32 days = 5 } #Unrest in the Baltics
			}
			15 = {
				country_event = { id = rus.33 days = 5 } #Unrest in Belarus
			}
			15 = {
				country_event = { id = rus.34 days = 5 } #Unrest in Ukraine
			}
			15 = {
				country_event = { id = rus.35 days = 5 } #Unrest in Transcaucasia
			}
			30 = {
				country_event = { id = rus.37 days = 5 } #Imperial Reforms
				
			}
		}
	}
}
